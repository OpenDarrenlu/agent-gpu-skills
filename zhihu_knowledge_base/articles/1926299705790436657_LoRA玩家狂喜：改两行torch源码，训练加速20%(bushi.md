附修改后的 ForCausalLMLoss ，自行替换 torch 中的实现：

def ForCausalLMLoss(
    logits,
    labels,
    vocab_size: int,
    num_items_in_batch: Optional[int] = None,
    ignore_index: int = -100,
    shift_labels: Optional[torch.Tensor] = None,
    **kwargs,
) -> torch.Tensor:
    # Upcast to float if we need to compute the loss to avoid potential precision issues
    logits = logits.float()

    if shift_labels is None:
        # Shift so that tokens < n predict n
        labels = nn.functional.pad(labels, (0, 1), value=ignore_index)
        shift_labels = labels[..., 1:].contiguous()

    # Flatten the tokens
    logits = logits.view(-1, vocab_size)
    shift_labels = shift_labels.view(-1)
    # Enable model parallelism
    shift_labels = shift_labels.to(logits.device)

    # Create a mask to filter out ignore_index
    mask = shift_labels != ignore_index

    # Apply the mask to logits and labels
    filtered_logits = logits[mask]
    filtered_labels = shift_labels[mask]

    # Compute the loss only on the filtered elements
    reduction = "sum" if num_items_in_batch is not None else "mean"
    if filtered_labels.numel() > 0:
        loss = nn.functional.cross_entropy(filtered_logits, filtered_labels, reduction=reduction)
    else:
        # Handle the case where all labels are ignore_index
        loss = torch.tensor(0.0, device=logits.device, requires_grad=True)

    if reduction == "sum":
        loss = loss / num_items_in_batch
    # print(f'loss {loss}')
    return loss
