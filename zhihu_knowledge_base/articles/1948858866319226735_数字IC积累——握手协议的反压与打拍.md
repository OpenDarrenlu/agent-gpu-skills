握手协议

基本握手协议如下：sender为数据发送端，receiver为数据接收端。sender给receiver发送数据时，同时发送valid信号以表示该数据有效，receiver能接收数据时，给sender发出ready信号，只有valid信号和ready信号同时有效时，才表示sender的数据发给了receiver。

 o_send_valid && i_receive_ready

直接打拍的问题

常见问题:在实现时，可能会遇见sender和receiver之间的走线太长以至于无法满足时序的情况。

情况1：sender到receiver的data和valid时序不满足，但是receiver返回的ready时序满足；
情况2：sender到receiver的data和valid时序满足，但是receiver返回的ready时序不满足；
情况3：valid，data和ready时序都不满足

对于时序问题，最常见的解决方法就是打拍，但是如果对握手协议直接打拍则会出现数据丢失或者重复情况，下面分别分析。

对valid和data打拍

对应情况1，在valid和data路径添加寄存器。

握手协议打拍方案

上述3中情况出错的原因是：发送端和接收端的握手信号不一致，两边不再是即时同步。为了解决这个问题，需要保证所插入的打断时序的寄存器和两端都有握手。如图所示，中间插入buffer，和发送端、接收端均握手处理。

两端都握手
方案1

该方案对valid和data进行了打拍，但是ready仍然是组合逻辑。

module buffer #(
    DWIDTH    = 32
)(
    input                     i_clk,
    input                     i_rst_n,

    input                     i_send_valid,
    input [DWIDTH-1 : 0]      i_send_data ,
    output                    o_send_ready,

    output                    o_receive_valid,
    output[DWIDTH-1 : 0]      o_receive_data ,
    input                     i_receive_ready
);

    reg   [DWIDTH-1 : 0]      r_buf_data ;    // 用于存放数据
    reg                       r_buf_valid;    // 数据有效

    // r_buf_valid
    always@(posedge i_clk or negedge i_rst_n)begin
        if(！i_rst_n)begin
            r_buf_valid <= 1'b0;
        end else if( i_send_valid ) begin                     // send端有数据请求发进来
            r_buf_valid <= 1'b1;
        end else if( ~i_send_valid && i_receive_ready ) begin  // receive端想接收，但是send不能发数据
            r_buf_valid <= 1'b0;
        end
    end
    
    // r_buf_data
    always@(posedge i_clk)begin
        if( i_send_valid && o_send_ready)) begin   // 当send端握手成功
            r_buf_data <= i_send_data;
        end
    end 
   
    
    assign o_send_ready = ~r_buf_valid || i_receive_ready;  //buf为空或者buf的数据被receive端接收（马上空）
    assign o_receive_valid = r_buf_valid;
    assign o_receive_data  = r_buf_data ;
    

endmodule

方案2

该方案对valid，data以及ready都进行了打拍，需要注意，当ready打拍后，如果buffer中只有一个数据空间，则如果T0时发送端的数据放进去，即使T1将buffer数据给到接收端，ready信号也要在T2才能拉起来，这样需要T2才能继续接收发送端数据，也就是产生了气泡。为了提高效率消除气泡，需要至少两个数据存放的空间。

//=====================================================================
// data flow
// i_send_data -> r_buf_data[1] -> r_buf_data[0] -> o_receive_data
//     |                                ^
//     |                                |
//      —— —— —— —— —— —— —
//======================================================================

module buffer #(
    DWIDTH    = 32
)(
    input                     i_clk,
    input                     i_rst_n,

    input                     i_send_valid,
    input [DWIDTH-1 : 0]      i_send_data ,
    output                    o_send_ready,

    output                    o_receive_valid,
    output[DWIDTH-1 : 0]      o_receive_data ,
    input                     i_receive_ready
);

    reg   [DWIDTH-1 : 0]      r_buf_data [1:0];    // 用于存放数据
    reg                       r_buf_valid[1:0];    // 数据有效信号
                                       

    // r_buf_valid[0] 
    always@(posedge i_clk or negedge i_rst_n)begin
        if(!i_rst_n)begin
            r_buf_valid[0] <= 'h0;
        end else if(r_buf_valid[1] || i_send_valid) begin //只要sender有值要进来或data1有效，data0就有数
            r_buf_valid[0] <= 'h1;
        end else if(i_receive_ready && ~r_buf_valid[1] && ~i_send_valid)begin //有数出去没数进来
            r_buf_valid[0] <= 'h0;
        end
    end

    // r_buf_data[0]
    always@(posedge i_clk)begin
        if(r_buf_valid[1] && i_receive_ready)begin // 若第一个数出去，且第二个位置有数
            r_buf_data[0] <= r_buf_data[1];
        end else if(~r_buf_valid[1] && i_send_valid) begin// 第二个位置没数但发送端数据有效
            if(~r_buf_valid[0] || i_receive_ready) begin //第一个位置没数或即将没数
                r_buf_data[0] <= i_send_data;
            end
        end

    // r_buf_valid[1]
    always@(posedge i_clk or negedge i_rst_n)begin
        if(!i_rst_n)begin
            r_buf_valid[1] <= 'h0; 
        end else if(r_buf_valid[0] && i_send_valid && (r_buf_valid[1]||~i_receive_ready)) begin 
        // 第一个位置有值，且有进无出或者有进有出但第二个数也有值
            r_buf_valid[1] <= 'h1;
        end else if(i_receiver_ready && ~i_send_valid)begin // 有出无进
            r_buf_valid[1] <= 'h0;
        end

    // r_buf_data[1]
    always@(posedge i_clk)begin
        if(i_send_valid) begin // 发送端数有效
           if(r_buf_valid[1] && r_buf_valid[0] && i_receive_ready ) begin // 流水
                r_buf_data[1] <= i_send_data;
           end else if (~r_buf_valid[1] && r_buf_valid[0] && ~i_receive_ready ) // 填充
                r_buf_data[1] <= i_send_data;
           end
        end
    end

    assign o_receive_valid = r_buf_valid[0];
    assign o_receive_data  = r_buf_data [0];

    // o_send_ready 寄存器触发条件为下一拍两个buffer空间不会满
    always@(posedge i_clk or negedge i_rst_n) begin
        if(!i_rst_n)begin
            o_send_ready <= 'h1;     // 初始有效
        end else if((~r_buf_valid[0]) || i_receiver_ready || (~r_buf_valid[1]&&~i_send_valid))begin 
       // 1、如果有两个buf空间，则下一拍不会满。2.接收端能接收，则下一拍不会满。3、还有空间且不进数，则下一拍不会满
            o_send_ready <= 'h1; 
        end else if(r_buf_valid[0] && i_send_valid && ~i_receive_ready)begin//只剩一个空间，且有进无出就要满
            o_send_ready <= 1'b0;
        end
    end

endmodule
