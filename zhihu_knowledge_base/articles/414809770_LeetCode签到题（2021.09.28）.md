/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     struct TreeNode *left;
 *     struct TreeNode *right;
 * };
 */

#define MAX_LEN 2000

void DFS(struct TreeNode *root, int targetSum, int *preSum, int depth, int *num) {
    //空节点返回
    if (root == NULL) {
        return;
    }
    //计算前缀和
    preSum[depth] = preSum[depth - 1] + root->val;
    //判断是否有路径
    for (int i = 0; i < depth; i++) {
        if (preSum[depth] - preSum[i] == targetSum) {
            (*num)++;
        }
    }
    //深入左右子树  
    depth++;
    struct TreeNode *leftchild = root->left;
    DFS(leftchild, targetSum, preSum, depth, num);
    struct TreeNode *rightchild = root->right;
    DFS(rightchild, targetSum, preSum, depth, num);
    preSum[depth] = 0;
    depth--;
    return;
}

int pathSum(struct TreeNode* root, int targetSum){
    //空节点返回
    if (root == NULL) {
        return 0;
    }
    int res = 0;
    int preSum[MAX_LEN] = {0};
    preSum[0] = 0;
    int depth = 1;
    DFS(root, targetSum, preSum, depth, &res);
    return res;
}
