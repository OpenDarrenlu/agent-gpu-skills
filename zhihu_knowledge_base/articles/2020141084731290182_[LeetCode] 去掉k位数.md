class Solution:
 def removeKdigits(self, num: str, k: int) -> str:
        mono = []
        for c in num:
            while k and mono and int(mono[-1]) > int(c):
                mono.pop()
                k -= 1
            # do not start with 0
            if len(mono) == 0 and c == '0':
                continue
            mono.append(c)
        # 还有就接着删
        while k and mono:
            mono.pop()
            k -= 1
        return ''.join(mono) if mono else '0'
