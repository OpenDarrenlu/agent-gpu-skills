8 系统监视器

当我们说，要做一个系统监视器的时候，我们会想到什么？Cpu占用率、内存使用量、硬盘使用量。我们可能还想要监控网速、进程数目、服务器温度、系统版本。在linux上，这些很容易实现，cat /proc 就能实现大多数。而在nodejs上存在模块child_process,也就是说，我们完全具备调用shell命令获取执行结果并整理数据的可能。

一开始，我选择命令的方案是$(uname -a)获取系统版本信息，$(lscpu)获取CPU信息，$(cat /pto/stat) 获取cpu使用量，再通过计算获取CPU“瞬时”使用率，$(free)获取内存信息，$(df -h)获取硬盘信息， $(ps -aux)获取进程信息，$(uptime)获取启动时间，$(cat /sys/class/thermal/thermal_zone0/temp)获取温度信息，$(ifconfig)获取网络信息。总之，利用这些常用或不常用的命令，我们可以获取到我们想要的基本信息。我们设想前端每1s发送一次请求，然后我们作出回应。尽管我们的做法是异步的，也就是说前端请求然后我们发给它的数据可能是1s钟之前的，但我们相信这1s钟并不会发生什么大事，因此不需要等所有信息都收集到再发送出去（就像我们再readIndex的时候所做的那样），只是每次发送出去后，运行一下我们的程序更新一下信息即可，以等待前端下次取用。所以，按照这种思路，我们可以写下大致的框架：

返回一个占用率（百分数）、一个主要信息（title）、一个详细信息（直接贴出原来的输出）即可。对于CPU占用率，应使用当次获得的CPU总忙碌时间减去上一次CPU总忙碌时间再除以两次测得的CPU总运行时间差来计算。对于网卡收发速率，应使用两次网卡收/发总Byte量除以两次真实时间间隔来计算。

就这样，我们写完了所有的获取信息的函数，并在前端使用ajax设定时间间隔1s发送一次请求信息，并把信息不做处理直接打印在屏幕上。看到信息全部都按照我们想要的方式呈现，开心地以为自己已经成功了，可不曾想，打开任务管理器，这nodejs竟一度飙升到10%的cpu占用率（Intel(R) Core(TM) i5-6300U CPU @ 2.40GHz, wsl1）！

就因为这个？10%？合着我写一个监视器反倒会占用我服务器宝贵的计算资源，那我将来想要部署其它服务怎么办？

为了探明原因，我使用console.time()和console.timeEnd()对分别排查每个函数的执行时间（在函数开始执行时放console.time()，在exec回调函数中放console.timeEnd()），发现情形不容乐观。但有一个规律就是，使用cat命令比其它命令更节约时间。

一般在这种情况下，正常思路就是能用cat某文件获取的信息就不用其他命令，这样把代码修改一遍。但我当时突发奇想，并没有先进行这一工作，而是想cat就是读取内容到stdout这一简单过程，而我们要的也只是文本的内容而已。既然如此，我们为何不用fs.readFile替代cat呢,要知道fs可是原生的东西，而cat是需要nodejs先创建子进程、调用shell、执行文件读取、获得数据如此多的步骤才能获得那一点东西的命令。这么想来，我用fs.readFile先替代了一个含cat命令的函数，果然，cpu占用率产生明显下降。接着，我便开始把能用读取文件替代的获取信息的渠道全部改为用读取文本来实现，最终，cpu占用率从10%降到2%。

2%，对于系统监视器来说确实是个不错的成绩，但，看看我设计的监视器的功能，和任务管理器/gnome-monitor的功能，额，2%，很差啊。

先不管这个，在实际中还遇到的bug就是万一存在两个人同时用浏览器查看同一台服务器的信息，那请求就不再是1s一次了，要是多个网页同时请求查看服务器信息，系统监视器的CPU占用率将成倍上升，到时候又飙回10%，之前的努力全部白费。

针对上述问题，我的思路是限定函数执行时间间隔，若两次请求的时间间隔低于某个时间（如1s）那就不执行函数，直接返回之前测得的信息。这样，无论存在多少用户，cpu占用率能始终在单用户（每隔1s请求一次）的水平，或者0%（没有用户请求信息）。另外，1s对于网页监视器来说也有点短，干脆改成3s, 这样，node的CPU占用率终于被控制在1%以下。

/*
 * use fs, exec, (except for those as parameters)
 * written by Mzero for MikoSite, MIT License
 * return system info = {cpuInfo, memInfo, diskInfo, process, network, temperature, uptime, version}  
 */
 
var fs = require('fs');
var exec = require('child_process').exec;
 
module.exports = function monitor() {
    // InDependent=========================================//
    function TimeHuman(time) {
        var result = "";
        var time_s = Math.round(time);
        var time_m = Math.floor(time_s / 60);
        var time_h = Math.floor(time_m / 60);
        var time_d = Math.floor(time_h / 24);
        time_s = time_s - time_m * 60;
        time_m = time_m - time_h * 60;
        time_h = time_h - time_d * 24;
        result = time_d + ' : ' + time_h + ' : ' + time_m + ' : ' + time_s;
        return result;
    }
 
    var cpuInfo = { "percent": "", "title": "", "Idle": 0, "Total": 0 };
    function getCpu() {
        fs.readFile("/proc/cpuinfo", function (err, data) {
            if (!err) {
                var info = data.toString().split('\n');
                var name = [];
                for (var i in info) {
                    if (info[i].indexOf('model name') != -1) {
                        var tmp = info[i].substring(info[i].lastIndexOf(':') + 1);
                        if (name.indexOf(tmp) == -1) name.push(tmp);
                    }
                }
                cpuInfo.title = name.toString();
            }
        });
        fs.readFile("/proc/stat", function (err, data) {
            if (!err) {
                var info = data.toString().split('\n');
                info = info.filter(function (s) { return s && s.trim(); });
                info = info[0].split(' ').filter(function (s) { return s && s.trim(); });
                var tmpTotal = 0, tmpIdle = Number(info[4]);
                for (var i = 1; i < info.length; i++)
                    tmpTotal = tmpTotal + Number(info[i]);
                if (tmpIdle == cpuInfo.Idle || tmpTotal == cpuInfo.Total) return;
                cpuInfo.percent = 100 - 100 * (tmpIdle - cpuInfo.Idle) / (tmpTotal - cpuInfo.Total);
                cpuInfo.Idle = tmpIdle; cpuInfo.Total = tmpTotal;
            }
        });
    }
    var memInfo = { "percent": "", "title": "", "output": "" };
    function getMem() {
        fs.readFile("/proc/meminfo", function (err, data) {
            if (!err) {
                var info = data.toString().split('\n');
                info = info.filter(function (s) { return s && s.trim(); });
                var MemTotal = info[0].split(' ').filter(function (s) { return s && s.trim(); });
                var MemFree = info[1].split(' ').filter(function (s) { return s && s.trim(); });
                MemTotal = Number(MemTotal[1]) >> 10;
                MemFree = Number(MemFree[1]) >> 10;
                memInfo.percent = 100 - 100 * MemFree / MemTotal;
                memInfo.title = "Total: " + String(MemTotal) + "MB | Used: "
                    + String(MemTotal - MemFree) + "MB | Free: " + String(MemFree) + "MB";
                memInfo.output = data.toString();
            }
        });
    }
    var diskInfo = { "percent": "", "title": "", "output": "" };
    function getDisk() {
        exec("df -h 2>&1", function (err, stdout, stderr) {
            if (!err) {
                var info = stdout.split('\n').filter(function (s) { return s && s.trim(); });
                for (var i in info) {
                    var infoDisk = info[i].split(' ').filter(function (s) { return s && s.trim(); });
                    if (infoDisk.indexOf('/') == 5) {
                        diskInfo.percent = Number(infoDisk[4].substring(0, infoDisk[4].indexOf('%')));
                        diskInfo.title = 'Usage of / : ' + infoDisk[4];
                        diskInfo.output = stdout;
                        break;
                    }
                }
            }
        });
    }
    var process = { "title": "", "output": "" };
    function getProc() {
        exec("ps -aux 2>&1", function (err, stdout, stderr) {
            if (!err) {
                var info = stdout.split('\n');
                info = info.filter(function (s) { return s && s.trim(); });
                var infoProc = { "User": [], "Num": [] };
                var TotalProc = 0;
                for (var i = 1; i < info.length; i++ , TotalProc++) {
                    var User = info[i].trim();
                    User = User.substring(0, User.indexOf(' '));
                    if (infoProc.User.indexOf(User) == -1) {
                        infoProc.User.push(User);
                        infoProc.Num.push(1);
                    } else {
                        infoProc.Num[infoProc.User.indexOf(User)]++;
                    }
                }
                process.title = 'Total: ' + TotalProc + ' | ';
                for (var i = 0; i < infoProc.User.length; i++) {
                    process.title += infoProc.User[i] + ': ' + infoProc.Num[i];
                    if (i != infoProc.User.length - 1) process.title += '     |     ';
                }
                process.output = stdout;
            }
        });
    }
    var network = []; // {"dev","ipv4","ipv6","RxSpeed","TxSpeed","RxBytes","TxBytes","time"}
    function getNetwork() {
        exec("ifconfig 2>&1", function (err, stdout, stderr) {
            if (!err) {
                var info = stdout.split('\n\n').filter(function (s) { return s && s.trim(); });
                for (var i = 0; i < info.length; i++) { // devices
                    var date = new Date();
                    // console.log("lastRx", lastRxTime, "lastTx", lastRxTime, "Now", timeNow);
                    var infoNet = {
                        "dev": "", "ipv4": "", "ipv6": "", "RxSpeed": 0, "TxSpeed": 0,
                        "RxBytes": 0, "TxBytes": 0, "time": date.getTime()
                    }
                    infoNet.dev = info[i].substring(0, info[i].indexOf(":"));
                    var subInfo = info[i].split('\n').filter(function (s) { return s && s.trim(); });
                    for (var k in subInfo) {
                        if (subInfo[k].indexOf('inet ') != -1) {
                            var tmp = subInfo[k].split(' ').filter(function (s) { return s && s.trim(); });
                            infoNet.ipv4 = tmp[1].toString();
                        }
                        if (subInfo[k].indexOf('inet6') != -1) {
                            var tmp = subInfo[k].split(' ').filter(function (s) { return s && s.trim(); });
                            infoNet.ipv6 = tmp[1].toString();
                        }
                        if (subInfo[k].indexOf('RX packets') != -1) {
                            var tmp = subInfo[k].split(' ').filter(function (s) { return s && s.trim(); });
                            infoNet.RxBytes = Number(tmp[4]);
                        }
                        if (subInfo[k].indexOf('TX packets') != -1) {
                            var tmp = subInfo[k].split(' ').filter(function (s) { return s && s.trim(); });
                            infoNet.TxBytes = Number(tmp[4]);
                        }
                    }
                    var lastRx = 0, lastTx = 0, lastTime = 0;
                    if (network[i]) {
                        let tmp = network[i];
                        lastRx = (tmp.RxBytes) ? tmp.RxBytes : 0;
                        lastTx = (tmp.TxBytes) ? tmp.TxBytes : 0;
                        lastTime = (tmp.time) ? tmp.time : 0;
                    }
                    if (infoNet.time) {
                        infoNet.RxSpeed = Math.round((infoNet.RxBytes - lastRx) / (infoNet.time - lastTime));
                        infoNet.TxSpeed = Math.round((infoNet.TxBytes - lastTx) / (infoNet.time - lastTime));
                    }
                    network[i] = infoNet;
                }
                var tmpIfconf = [];
                for (var i = 0; i < network.length; i++)
                    if (network[i].dev != "lo") tmpIfconf.push(network[i]);
                network = tmpIfconf;
            }
        });
    }
    var version = "";
    function getVersion() {
        fs.readFile("/proc/version", function (err, data) {
            if (!err) {
                data = data.toString();
                version = data.substring(0, data.indexOf("#"));
            }
        });
    }
    var uptime = "";
    function getUptime() {
        fs.readFile("/proc/uptime", function (err, data) {
            if (!err) {
                var stdout = data.toString();
                var info = stdout.split(' ');
                info = info.filter(function (s) { return s && s.trim(); });
                uptime = TimeHuman(Number(info[0]));
            }
        });
    }
    var temperature = "";
    function getTemp() {
        fs.readFile("/sys/class/thermal/thermal_zone0/temp", function (err, data) {
            if (!err) temperature = String(Number(data.toString()) / 1000) + " tC";
        });
    }
 
    // Methods ======================================================//
    var cTime = 0;
    var minAskTime = 3000;
    this.getInfo = function () {
        // if ask twice in 3 seconds, return same value 
        var date = new Date();
        var tTime = date.getTime();
        if (tTime - cTime > minAskTime) {
            cTime = tTime;
            getCpu(); getMem(); getDisk();
            getNetwork(); getProc();
            getVersion(); getUptime(); getTemp();
        }
 
        // var cpuInfo = { "percent": "", "title": "", "Idle": 0, "Total": 0 };
        // var memInfo = { "percent": "", "title": "" };
        // var diskInfo = { "percent": "", "title": "", "output": "" };
        // var process = { "title": "", "output": "" };
        // var network = []; // {"dev", "ipv4", "ipv6","RxSpeed", "TxSpeed", "RxBytes", "TxBytes", "time"}
        // var temperature = "", uptime = "", version = "";
 
        return JSON.stringify({
            cpuInfo: cpuInfo,
            memInfo: memInfo,
            diskInfo: diskInfo,
            process: process,
            network: network,
            version: version,
            uptime: uptime,
            temperature: temperature
        });
    }
}

这里特别要提一下那个环形进度条的实现。为使用原生js实现环形进度条，我搜了不少网页，终于找到一个靠谱的方法：

首先，中间的区域并不是透明的，而是强行贴了一块白色圆形（或者说是border-radius:50%的正方形div）。另外，背底的颜色其实才是进度条的颜色。为解释其具体的实现过程，我们不妨先拿掉中间的圆形片。那么我们的进度条是这样的：

左半圆（橙色），右半圆（黄色），背底圆（蓝色）。可以想象，当右半圆顺时针旋转时，蓝色露出的部分就会变多，这样就能表示0-50%的进度了。可是50%-100%该怎么实现呢？其实也很简单，只要让右半圆回到原位，并且让右半圆的颜色为蓝色，旋转左半圆，这样就行了。由于我们设计的时候左半圆在右半圆的下方，所以左半圆无论怎么转，右半边永远是右半圆的颜色，也就是背底的颜色，蓝色。




当然也可以右半圆回到原位变成蓝色后继续顺时针旋转，这个方案我还没有尝试过。

如果使左右半圆初始颜色相同，再贴上白色的圆片，那么我们就能看到进度条的模样了：

怎么样，有点进度条的感觉了吧。只要根据获得的数值（例如50）计算左右半圆需要旋转的角度便可以完成进度条了。另外，简易的动画效果可以由如下方案实现：




获得上一次数值和本次数值，初始将上一次数值定为当前进度，之后每隔一定时间如8ms，将当前进度加一或减一，直到当前进度与本次数值相一致停止动作。这样，我们的进度条就能一点一点地从上次的数据挪到本次数据了，这样便产生了动画效果。
