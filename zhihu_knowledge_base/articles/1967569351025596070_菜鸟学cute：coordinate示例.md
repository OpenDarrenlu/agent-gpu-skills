cutlass GTC一个talk里面有关hierarchical layout例子，最近开始cute编程写个小程序计算logical coordinates 1D,2D,N-D

代码如下：

#include <cute/layout.hpp>

using namespace cute;
using X = Underscore;

int main()
{
  auto my_shape = make_shape(make_shape(Int<2>{}, make_shape(Int<2>{}, Int<2>{})), make_shape(Int<2>{}, make_shape(Int<2>{}, Int<2>{})));
  auto my_stride = make_stride(make_stride(Int<1>{}, make_stride(Int<4>{}, Int<16>{})), make_stride(Int<2>{}, make_stride(Int<8>{}, Int<32>{})));
  auto my_layout = make_layout(my_shape, my_stride);

  printf("my_layout is:\n");
  print(my_layout);
  printf("\n");

  auto ur_shape = make_shape(make_shape(Int<2>{}, Int<4>{}), make_shape(Int<2>{}, Int<4>{}));

  printf("ur_shape is:\n");
  print(ur_shape);
  printf("\n");

  for (int i=0; i<size(my_shape); i++) {
     auto flat_coord_1 = my_layout.get_flat_coord(i);
     auto hier_coord_1 = my_layout.get_hier_coord(i);
     auto hier_coord_2 = crd2crd(hier_coord_1, my_shape, ur_shape);
     printf("i %2d flat-coord ", i);
     print(flat_coord_1);
     printf(" hier-coord ");
     print(hier_coord_1);
     printf(" hier-coord-2 ");
     print(hier_coord_2);
     printf("\n");
  }

  return 0;
}
