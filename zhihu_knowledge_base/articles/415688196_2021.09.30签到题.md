int computeArea(int ax1, int ay1, int ax2, int ay2, int bx1, int by1, int bx2, int by2){
    int res = 0;
    int aArea = (ax2 - ax1) * (ay2 - ay1);
    int bArea = (bx2 - bx1) * (by2 - by1);
    int x = 0;
    int y = 0;
    x = fmax(ax1, bx1) < fmin(ax2, bx2) ? (fmin(ax2, bx2) - fmax(ax1, bx1)) : 0;
    y = fmax(ay1, by1) < fmin(ay2, by2) ? (fmin(ay2, by2) - fmax(ay1, by1)) : 0;
    res = aArea + bArea - x * y;
    return res;
}
