基本算法：k 维度的循环分块：

一个thread block完成C矩阵中的一个小块（TileC）的计算任务 (由 TiledMMA 定义计算方式)，如下图中的（bm, bn）绿色方块。

通过block维度的扩展，如图所示的矩阵C中M轴方向的blockIdx.y和N轴方向的blockIdx.x的扩展，便可以完成整个C矩阵的计算。由此我们可以计算出完成整个C矩阵所需要的gird维度：

grid.x = N / bN
grid.y = M / bM // (此处暂时不考虑不可整除的情形)。

gemm_v1：最基础的版本

直接按照分块的思想，将每个分块拷贝到寄存器中，然后直接使用 MMA 指令计算。

host 端和device 端代码如下：

template<typename T, int bM, int bN, int bK, typename TiledMMA>
__global__ void gemm_device(const T* Aptr, const T* Bptr, T* Cptr, 
                            int m, int n, int k) {
  using namespace cute;
  using TA = float;
  using TB = float;

  Tensor A = make_tensor(make_gmem_ptr(Aptr), make_shape(m, k), make_stride(k, Int<1>{})); //(m,k) row-major
  Tensor B = make_tensor(make_gmem_ptr(Bptr), make_shape(n, k), make_stride(k, Int<1>{})); //(n,k) row-major
  Tensor C = make_tensor(make_gmem_ptr(Cptr), make_shape(m, n), make_stride(n, Int<1>{})); //(m,n) row-major

  // Get the appropriate blocks for this thread block
  int ix = blockIdx.x;
  int iy = blockIdx.y;             
  Tensor gA = local_tile(A, make_tile(Int<bM>{}, Int<bK>{}), make_coord(iy, _));  // (b_M,b_K,num_tile_k)
  Tensor gB = local_tile(B, make_tile(Int<bN>{}, Int<bK>{}), make_coord(ix, _));  // (b_N,b_K,num_tile_k)
  Tensor gC = local_tile(C, make_tile(Int<bM>{}, Int<bN>{}), make_coord(iy, ix)); // (b_M,b_N)
  
  TiledMMA tiled_mma;
  ThrMMA thr_mma = tiled_mma.get_thread_slice(threadIdx.x);
  Tensor tAgA = thr_mma.partition_A(gA); // (MMA, MMA_M, MMA_K, num_tile_k)
  Tensor tBgB = thr_mma.partition_B(gB); // (MMA, MMA_N, MMA_K, num_tile_k)
  Tensor tCgC = thr_mma.partition_C(gC); // (MMA, MMA_M, MMA_N)
  

  auto tArA = thr_mma.partition_fragment_A(gA(_, _, 0));  // (MMA, MMA_M, MMA_K)
  auto tBrB = thr_mma.partition_fragment_B(gB(_, _, 0));  // (MMA, MMA_K, MMA_N)
  auto tCrC = thr_mma.partition_fragment_C(gC(_, _));     // (MMA, MMA_M, MMA_N)

  clear(tCrC); //将寄存器中的数据初始化为0
  int num_tile_k = size<2>(gA);

  #pragma unroll 1
  for(int itile = 0; itile < num_tile_k; ++itile) {
    cute::copy(tAgA(_, _, _, itile), tArA);
    cute::copy(tBgB(_, _, _, itile), tBrB);

    cute::gemm(tiled_mma, tCrC, tArA, tBrB, tCrC);
  }

  cute::copy(tCrC, tCgC); 
}

template<typename T>
void gemm_v1(const T* A, const T* B, T* C, int M, int N, int K) {
  const int bM = 128;
  const int bN = 256;
  const int bK = 32;

  using mma_op = SM80_16x8x16_F16F16F16F16_TN; // A(M,K) row-major. B(K,N) col-major 等同于 B(N,K) row-major
  using mma_traits = MMA_Traits<mma_op>;
  using MMA_Atom_Arch = MMA_Atom<mma_traits>;

  static constexpr int kNWarps = 4;

  using TiledMMA = TiledMMA<MMA_Atom_Arch,
                      Layout<Shape<Int<kNWarps>,_1,_1>>,
                      Tile<Int<16 * kNWarps>, _16, _16>>;
//   using TiledMMA = decltype(make_tiled_mma(MMA_Atom_Arch{}, 
//                       make_layout(Shape<_2, _2, _1>{}), 
//                       make_layout(Shape<_4, _1, _1>{})));
  // 上面两种写法等同
//   print(TiledMMA{});
  dim3 dimGrid(size(ceil_div(N, bN)), 
               size(ceil_div(M, bM)));
  dim3 dimBlock(size(TiledMMA{}));

  gemm_device<T, bM, bN, bK, TiledMMA><<<dimGrid, dimBlock, 0, 0>>>(A, B, C,  M, N, K);
}

gemm_v2: 高效的内存访问

利用共享内存提高数据搬运的效率，实现高效的数据复用，从而提升提升访存的效率。

优化点如下：

1. gmem --> smem 时，使用向量化异步 copy 指令 SM80_CP_ASYNC_CACHEGLOBAL

gemm_v3: 尾阶段高效（Epilogue）

上面的demo中，每个线程直接将寄存器中的数据写回 global memory，每个线程中的寄存器数据并不多，这样就会有大量的低效的内存传输指令。因为优化的策略是，CTA中每个线程先把自己寄存器中的数据存放在一个公有的 shared memory 中，存到一定量时，再一起通过更高效的传输指令 copy 到gmem中。

先思考一个问题，每个线程有多少个寄存器存放多少个有效的计算结果。

以 SM80_16x8x16_F16F16F16F16_TN 指令集为例，warp 中32 个线程负责 C 中16x8 （m，n）个元素的计算，每个线程负责 4 个元素，根据指令集的示意图可以，这4个元素的shape 可以表示为 (2,2)

通过线程拓展和寄存器拓展，MMA 指令可以完成更大规模的 (mnk)矩阵计算，如下代码：

这里的寄存器拓展的说法，我的理解是warp level 重复。

    // mma
    using mma_op = SM80_16x8x16_F16F16F16F16_TN;
    using mma_traits = MMA_Traits<mma_op>;
    using mma_atom = MMA_Atom<mma_traits>;
    static constexpr int kMmaEURepeatM = 2;
    static constexpr int kMmaEURepeatN = 2;
    static constexpr int kMmaEURepeatK = 1;
    
    static constexpr int kMmaVRepeatM = 1;
    static constexpr int kMmaVRepeatN = 2;
    static constexpr int kMmaVRepeatK = 1;

    using mma_atom_shape = mma_traits::Shape_MNK;
    static constexpr int kMmaPM = get<0>(mma_atom_shape{}) * kMmaEURepeatM * kMmaVRepeatM;
    static constexpr int kMmaPN = get<1>(mma_atom_shape{}) * kMmaEURepeatN * kMmaVRepeatN;
    static constexpr int kMmaPK = get<2>(mma_atom_shape{}) * kMmaEURepeatK * kMmaVRepeatK;
    using MMA_EU_RepeatT = decltype(make_layout(make_shape(
        Int<kMmaEURepeatM>{}, Int<kMmaEURepeatN>{}, Int<kMmaEURepeatK>{})));
    using MMA_P_T = Tile<Int<kMmaPM>, Int<kMmaPN>, Int<kMmaPK>>;
    using MMA = decltype(make_tiled_mma(mma_atom{}, MMA_EU_RepeatT{}, MMA_P_T{}));


上述代码 kMmaVRepeatN = 2， 所以每个线程需要更多的寄存器，这里附上 read 大佬的这个图一起理解：

个人对这部分的理解是（不一定对……^_^，大佬们可以在平均区指点交流）

kMmaVRepeatM, kMmaVRepeatN, kMmaVRepeatK 定义了 warp level 重复，然后得到的 TiledMMA 会在目标tensor上进行周期性重复，这是 thread block level 重复。cute中二者都需要更多的寄存器。

定义完 TiledMMA 后，在 warp level 完成循环计算后，每个线程寄存器存的 C 中的一部分shape 为：

(2， (2*kMmaVRepeatM， 1*kMmaVRepeatN))

然后 TiledMMA 会根据目标tensor进行周期性平铺(thread level 循环迭代)，这也需要额外的寄存器，因此，每个线程的寄存器shape可以理解为：

((2， (2*kMmaVRepeatM， 1*kMmaVRepeatN)), bM/kMmaPM, bN/kMmaPN)

如上，（2，2）是warp每次循环需要的寄存器，(2， (2*kMmaVRepeatM， 1*kMmaVRepeatN) 是warp level 循环一共需要的寄存器，(2， (2*kMmaVRepeatM， 1*kMmaVRepeatN), bM/kMmaPM, bN/kMmaPN) 是完整的 thread block level 循环需要的全部寄存器。

对于 MMA 指令来说，不管是 warp level 还是 thread block level 循环，计算还是以warp为基本单元，每次要求的输入 tileC 部分的 shape 依然是 （2，2），因此，从这个角度上理解，每个线程的寄存器shape 为：

((2， 2), bM /（16 *kMmaEURepeatM）, bN / (8 * kMmaEURepeatN))

一般表示为：（MMA, MMA_M, MMA_N)

二者 size 是相等的。

kMmaEURepeatM * kMmaEURepeatN 个 warp （线程拓展）一次 thread block level 迭代完成C中（kMmaPM， kMmaPN）这么多元素的运算。因此，我们需要一块 （kMmaPM， kMmaPN）这么大的 shared memory。这时计算已经完成了，不必申请新的smem，而是直接服用 tileA or tileB 的shared memory。代码如下：

    // epilogue: register to global via shared memory
    auto kSmemLayoutCBatch = Int<2>{};
    using SmemLayoutAtomC = decltype(composition(
        Swizzle<3, 3, 3>{}, make_layout(make_shape(Int<kMmaPM>{}, Int<kMmaPN>{}),
                                        make_stride(Int<kMmaPN>{}, Int<1>{}))));
    using SmemLayoutC = decltype(tile_to_shape(
        SmemLayoutAtomC{},
        make_shape(Int<kMmaPM>{}, Int<kMmaPN>{}, Int<kSmemLayoutCBatch>{})));


这里多一个 kSmemLayoutCBatch， 可以一次性将多次 thread block level 迭代的结果存到 shared memory 后再进行传输，进一步提高访存效率。主要相关部分代码如下：

通过软件流水，可以 overlap 数据传输和计算耗时，这部分建议细度 read 大佬的这篇

核心逻辑代码和下面这个图完全对应：

图中每个同步点 or copy 发出点的对应位置，在下面代码中有标注。

                ismem_read = (ismem_read + 1) % kStage;
            }
             
            // 对应图中 S0,1 -> R1, S0,2 -> R2, S0,3 -> R3, S0,4 -> R4,
            // shm -> reg s[itile][ik + 1] -> r[ik + 1]
            cute::copy(s2r_tiled_copy_a, tAsA(_, _, ik_next, ismem_read), // tAsA: (CPY, CPY_M, CPY_K, kStage)
                    tCrA_view(_, _, ik_next));                            // tCrA_view: (CPY, CPY_M, CPY_K)
            cute::copy(s2r_tiled_copy_b, tBsB(_, _, ik_next, ismem_read), // tBsB: (CPY, CPY_M, CPY_K, kStage)
                    tCrB_view(_, _, ik_next));                            // tCrB_view: (CPY, CPY_M, CPY_K)
            
            // 对应图中左边第一条黑色实线
            if (ik == 0) {
                if (itile_to_read < ntile) {
                    cute::copy(g2s_tiled_copy_a, tAgA_copy(_, _, _, itile_to_read),
                            tAsA_copy(_, _, _, ismem_write));
                    cute::copy(g2s_tiled_copy_b, tBgB_copy(_, _, _, itile_to_read),
                            tBsB_copy(_, _, _, ismem_write));

                    ++itile_to_read;
                    ismem_write = (ismem_write + 1) % kStage;
                }

                cp_async_fence();
            }

            cute::gemm(tiled_mma, tCrD, tCrA(_, _, ik), tCrB(_, _, ik), tCrD);
        }  // for ik
    }
