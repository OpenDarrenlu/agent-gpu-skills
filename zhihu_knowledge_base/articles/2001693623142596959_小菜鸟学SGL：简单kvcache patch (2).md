            q = self.q_a_proj(hidden_states)[0]
            q = self.q_a_layernorm(q)
            q = self.q_b_proj(q)[0].view(-1, self.num_local_heads, self.qk_head_dim)
        else:
             ...

        # q vector contains 2 parts:q_nope and q_pe which is roped with positions
        # q_nope: uncompressed after q_b_proj (up projection)
        # here, just splits q_pe from q at the last dimension
        _, q_pe = q.split([self.qk_nope_head_dim, self.qk_rope_head_dim], dim=-1)


        # latent_cache includes 2 parts: compressed kv and k_pe roped with positions
        latent_cache = self.kv_a_proj_with_mqa(hidden_states)[0]
 

        # kv_a is compressed kv separated from latent cache
        # k shares compressed expression with v in latent cache
        # kv_a shape: [L, kv_lora_rank]
        kv_a, _ = latent_cache.split([self.kv_lora_rank, self.qk_rope_head_dim], dim=-1)

        # reshape latent_cache to [L, 1, kv_lora_rank + qk_rope_head_dim]
        latent_cache = latent_cache.unsqueeze(1)

        # kv is uncompressed kv after normalization and up projection (kv_b_proj)
        # kv shape: [L, num_local_heads * (qk_nope_head_dim + qk_rope_head_dim)]
        kv_a = self.kv_a_layernorm(kv_a.contiguous())
        kv = self.kv_b_proj(kv_a)[0]

        # reshape kv to [L, num_local_heads, qk_nope_head_dim + qk_rope_head_dim]
        kv = kv.view(-1, self.num_local_heads, self.qk_nope_head_dim + self.v_head_dim)
        # k_nope: the uncompressed key after k up projection
        k_nope = kv[..., : self.qk_nope_head_dim]
        # v: the uncompressed value after v up projection
        v = kv[..., self.qk_nope_head_dim :]

        # k_pe is retrieved directly from latent cache
        k_pe = latent_cache[:, :, self.kv_lora_rank :]

        # rope operation with positions along with q_pe and k_Pe
        q_pe, k_pe = self.rotary_emb(positions, q_pe, k_pe)

        # concat q_nope with q_pe
        q[..., self.qk_nope_head_dim :] = q_pe

        # concat k_nope with k_pe
        k = torch.empty_like(q)
        k[..., : self.qk_nope_head_dim] = k_nope
        k[..., self.qk_nope_head_dim :] = k_pe


        # actually latent cache is used to store compressed kv (after normalization) and k_pe
        # the compressed kv is necessary to restore k/v when doing incremental decoding, since at that time,
        # prefix prefill is handled by forward_absorb..SAD!
        # latent cache could be called as context
        latent_cache[:, :, : self.kv_lora_rank] = kv_a.unsqueeze(1)
        latent_cache[:, :, self.kv_lora_rank :] = k_pe

        # Save latent cache
        forward_batch.token_to_kv_pool.set_kv_buffer(
            self.attn_mha, forward_batch.out_cache_loc, latent_cache, None
        )
        attn_output = self.attn_mha(q, k, v, forward_batch, save_kv_cache=False)
        attn_output = attn_output.reshape(-1, self.num_local_heads * self.v_head_dim)
        output, _ = self.o_proj(attn_output)
        return output


forward_absorb

简单来说就是把kv_b_proj拆分成w_kc和w_vc

矩阵乘法满足结合律，把w_kc应用提前到attention计算之前，把w_vc应用延后到attention计算之后。这样，attention计算前，把uncompressed Q投影到kv的压缩空间，attention计算可以直接在kv的压缩空间执行。

但是，Q的attention heads是128，KV的attention heads是1

把k的up projection（w_kc)合并到Q：q_nope_out = torch.bmm(q_nope.transpose(0, 1), self.w_kc)，对于long extent场景权重计算在整体计算量上占比高，计算量会放大。w_vc效果类似。

        else:
            ...

        # apply w_kc to q_nope
        if self.w_kc.dtype == torch.float8_e4m3fnuz:
           ...
        elif self.w_kc.dtype == torch.float8_e4m3fn:
            ...
        else:
            q_nope_out = torch.bmm(q_nope.transpose(0, 1), self.w_kc)
        q_input[..., : self.kv_lora_rank] = q_nope_out.transpose(0, 1)

        latent_cache = self.kv_a_proj_with_mqa(hidden_states)[0]
        v_input = latent_cache[..., : self.kv_lora_rank]

        q_input[..., self.kv_lora_rank :] = q_pe
        k_input[..., self.kv_lora_rank :] = k_pe

        attn_output = self.attn_mqa(q_input, k_input, v_input, forward_batch)
        attn_output = attn_output.view(-1, self.num_local_heads, self.kv_lora_rank)

        # apply w_vc to attention output
        if self.w_vc.dtype == torch.float8_e4m3fnuz:
            ...
        elif self.w_vc.dtype == torch.float8_e4m3fn:
            ...
        else:
            attn_bmm_output = torch.bmm(attn_output.transpose(0, 1), self.w_vc)
        attn_output = attn_bmm_output.transpose(0, 1).flatten(1, 2)
        output, _ = self.o_proj(attn_output)

        return output
