代码解析
# 初始化
Dmem = [0]*1024
Imem = ['']*1024
reg = [0]*15
esp = 0
CF = ZF = SF = OF = 0

# 读入指令到存储器，需要判断指令类型来读取相应长度的机器码
def LoadInstr(path):
    global Imem
    f = open(path, 'r')
    line = f.readlines()[0]
    n = l = 0
    while len(line) > 0:
        k = int(line[0], 16)
        if k == 0:
            l = 2
        elif k == 1:
            l = 2
        elif k == 2:
            l = 4
        elif k == 3:
            l = 20
        elif k == 4:
            l = 20
        elif k == 5:
            l = 20
        elif k == 6:
            l = 4
        elif k == 7:
            l = 18
        elif k == 8:
            l = 18
        elif k == 9:
            l = 2
        elif k == 10:
            l = 4
        elif k == 11:
            l = 4
        Imem[n] = line[:l]
        n += 1
        line = line[l:]

# 由于Y86采用小端存储，tonum函数将该方式表示的16进制数转化为对应的10进制数
def tonum(s):
    l = len(s)
    num = ''
    for i in range(l//2):
        num += s[-2:]
        s = s[:-2]
    return int(num, 16)

# 条件码（本处理器未使用进位标志），每次运算后更新
def concode(num, a, b):
    global CF, ZF, SF, OF
    ZF = num == 0
    SF = num < 0
    OF = (a < 0 == b < 0) and (num < 0 != a < 0)

# 本处理器名为trash
def trash():
    global Dmem, Imem, reg, esp, ZF, SF, OF
    PC = 0
    while True:
        # 取指
        instr = Imem[PC]
        PC += 1

        # 译码&执行&存储
        code = int(instr[0], 16)
        func = int(instr[1], 16)
        if code == 0:
            print('HALT~')
            break
        elif code == 1:
            n += 1
            continue
        elif code == 2:
            if func == 0:
                reg[int(instr[3], 16)] = reg[int(instr[2], 16)]
            elif func == 1 and ((SF ^ OF) | ZF):
                reg[int(instr[3], 16)] = reg[int(instr[2], 16)]
            elif func == 2 and (SF ^ OF):
                reg[int(instr[3], 16)] = reg[int(instr[2], 16)]
            elif func == 3 and (ZF):
                reg[int(instr[3], 16)] = reg[int(instr[2], 16)]
            elif func == 4 and ((~ZF)):
                reg[int(instr[3], 16)] = reg[int(instr[2], 16)]
            elif func == 5 and (~(SF ^ OF)):
                reg[int(instr[3], 16)] = reg[int(instr[2], 16)]
            elif func == 6 and (~(SF ^ OF) & (~ZF)):
                reg[int(instr[3], 16)] = reg[int(instr[2], 16)]
        elif code == 3:
            reg[int(instr[3], 16)] = tonum(instr[4:20])
        elif code == 4:
            Dmem[reg[int(instr[3], 16)] + tonum(instr[4:20])] = reg[int(instr[2], 16)]
            print('r->m: %d' %(Dmem[reg[int(instr[3], 16)] + tonum(instr[4:20])]))
        elif code == 5:
            reg[int(instr[2], 16)] = Dmem[reg[int(instr[3], 16)] + tonum(instr[4:20])]
        elif code == 6:
            rA = int(instr[2], 16)
            rB = int(instr[3], 16)
            a = reg[rA]
            b = reg[rB]
            if func == 0:
                reg[rA] = reg[rA] + reg[rB]
            elif func == 1:
                reg[rA] = reg[rA] - reg[rB]
            elif func == 2:
                reg[rA] = reg[rA] & reg[rB]
            elif func == 3:
                reg[rA] = reg[rA] ^ reg[rB]
            concode(reg[rA], a, b)
        elif code == 7:
            if func == 0:
                PC = tonum(instr[2:18])
            elif func == 1 and ((SF ^ OF) | ZF):
                PC = tonum(instr[2:18])
            elif func == 2 and (SF ^ OF):
                PC = tonum(instr[2:18])
            elif func == 3 and (ZF):
                PC = tonum(instr[2:18])
            elif func == 4 and ((~ZF)):
                PC = tonum(instr[2:18])
            elif func == 5 and (~(SF ^ OF)):
                PC = tonum(instr[2:18])
            elif func == 6 and (~(SF ^ OF) & (~ZF)):
                PC = tonum(instr[2:18])
        elif code == 8:
            PC = tonum(instr[2:18])
        elif code == 9:
            PC = Dmem[esp]
            esp += 1
        elif code == 10:
            esp += 1
            Dmem[esp] = reg[int(instr[2], 16)]
        elif code == 11:
            reg[int(instr[2], 16)] = Dmem[esp]
            esp -= 1

# RUN（Instr.trash是下面“指令”部分对应的文件）
LoadInstr('Instr.trash')
trash()
