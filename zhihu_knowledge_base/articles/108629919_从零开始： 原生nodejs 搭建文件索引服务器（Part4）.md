9 表单数据处理

这一栏，本来应该是用来讲文件上传的，但我想了想，似乎可能需要先单列一栏分析一下如何处理表单数据。

那表单数据长什么样呢？F12见真相：

------WebKitFormBoundaryKQ0sSbPtViX7OSCw
Content-Disposition: form-data; name="files"; filename="图片2.png"
Content-Type: image/png
 
 
------WebKitFormBoundaryKQ0sSbPtViX7OSCw--


并且，请求头中包含了这一项：

Content-Type: multipart/form-data; boundary=----WebKitFormBoundaryKQ0sSbPtViX7OSCw


所以整理一下，大致就是：

--BOUNDARY\r\nINFO\r\n\r\nCONTENT\r\n--BOUNDARY--\r\n


当含有多个文件的时候，就像这样：

--BOUNDARY\r\nINFO\r\n\r\nCONTENT\r\n--BOUNDARY\r\nINFO\r\n\r\nCONTENT....\r\n--BOUNDARY--\r\n


处理表单数据，一开始我参考了网上某贴的思路，对每个\r\n进行匹配，定为一行，要是在这一行中找到相应的信息，便可判定为不是CONTENT，否则将改行写入文件。这么做处理小文件当然可以，但一旦文件变大，就会占用过多的内存。而且另一方面，这么做效率很低——因为首先要缓存一行、然后要转成string判断。假设要上传一个很大的binary文件，但文件中有很多连续的13,10(\r\n的ascii码),那么这种方法就会十分耗时。

因此首先应当对读一行这件事进行改进。一开始我的改进方案如下：

一开始先取得boundary,然后采用状态机的方式对文件进行判断。若状态（chunckStatus）为0，便以行为单位寻找boundary,找到boundary后以行为单位寻找指定信息，也就是INFO部分。以出现一行没有信息为标志，转入状态2，跳过一个\n,同时为状态3文件写入做准备。状态3开始文件写入，找到一个\r\n便进入boundary匹配状态。若匹配不到boundary,则将缓存的行（subData）写入磁盘，否则跳转到状态1，准备读取下一个文件的INFO.

尽管思路清晰，但代码实际上有点混乱。最终效果也没有达成，在文件超过一定大小（如10MB）之后，文件读取便开始破碎。

在这个问题上，我认为很有可能是write操作占用太多时间导致chunk没法处理，因此又采取数组的方法将write操作全部替换为写入数组。在状态机外额外判断如果数组长度大于一定值（如4*1024）则使用write方法写入硬盘。

几天后，在刷知乎的时候，偶然看见喷JS的帖子，说仗着网络协议不改变，写了许多与网络甚至特定内核的浏览器相契合的“好用”的代码。这篇帖子喷JS到底怎么差我没看进去，倒是对“协议不更新”印象深刻。既然协议不会更新，那就意味着我可以写死代码。

我们再来观察一下formData:

--BOUNDARY\r\nINFO\r\n\r\nCONTENT\r\n--BOUNDARY\r\nINFO\r\n\r\nCONTENT....\r\n--BOUNDARY--\r\n


一定是“--BOUNDARY\r\n”+“INFO(内部存在\r\n)\r\n\r\n”+“CONTENT”+“\r\n--BOUNDARY\r\n”+...

因此我们不妨编写状态机：当状态为0时（寻找边界），找到“--BOUNDARY\r\n”则转入状态1（寻找INFO）;当状态为1时，将数据全部存入数组，直到找到“\r\n\r\n”，把存储的数组转成字符串push到Infos中，转入状态2（寻找新边界）；在状态2中找到“\r”则判断下一个是不是“\n”,下下一个是不是“-”，如果是了，判断连起来是不是“\r\n--BOUNDARY”,如果是则判断接下来是“\r\n”还是“--\r\n”,借此判断是否为结尾。

其中在状态2采取一步一步判断的目的是文件中找到“\r\n-”的概率较小，在前几步就能排除掉大多数干扰串，从而减少判断的运算量。

另外，之前代码结构不清晰的原因我认为在很大程度上是for循环中i++的干扰，因此改变循环结构，自加多少全部由每个状态中代码自行决定。为了减少对chunk的阻碍，我还参照multiparty的方法先把文件存成临时文件名的文件（我这里以当前时间毫秒数命名），待请求结束时，再由用户自行决定重命名。这种思路编写的代码如下所示：

/*
 * use fs, http(Request, Response), (except for those as parameters)
 * written by Mzero for MikoSite, MIT License
 * parseFormData and store file into writestream, can proc even ~GB(per file) files  
 * example: 
 * new (require("./formFileParser"))({ uploadDir: __dirname + "/test" })
 * .parse(request, (err, files) => {});
 */
 
var fs = require("fs");
 
module.exports = function formFileParser(opts) {
    // parse parameters
    if (!opts || !("uploadDir" in opts)) {
        console.log("no uploadDir found");
        return;
    }
    if (!("bufSize" in opts)) opts.bufSize = 4096;
    var uploadDir = opts.uploadDir, bufSize = opts.bufSize;
    // correct uploadDir
    try {
        if (!fs.statSync(uploadDir).isDirectory()) {
            console.log(uploadDir, "is not a directory");
            return;
        }
    } catch (err) {
        try {
            fs.mkdirSync(uploadDir, { recursive: true });
        } catch (err) {
            console.log("mkdir:", uploadDir, "error");
            return;
        }
    }
 
    const stateSearchBoundary = 0;
    const stateSearchFileInfo = 1;
    const stateSearchFileEnd = 2;
 
    // callback(err, files[])
    // formData: BOUNDARY\r\nINFO\r\n\r\nCONTENT\r\nBOUNDARY\r\n...BOUNDARY--\r\n
    this.parse = (request, callback) => {
        var cType = request.headers['content-type'];
        if (typeof cType == "undefined") {
            callback("Error: content-type can not be found in headers", []);
            return;
        } else {
            var boundaryPos = cType.lastIndexOf("boundary=");
            if (boundaryPos == -1) {
                callback("Error: boundary can not be found in headers", []);
                return;
            }
        }
 
        var boundaryBuf = Buffer.from("--" + cType.substring(boundaryPos + 9, cType.length));
        var state = stateSearchBoundary;
        var data = [], info = [], tmpFiles = [], subInfoArr = [];
        var writeStream = null;
        request.on('data', (chunck) => {
            // use chunk, boundaryBuf, subInfoArr, state
            function searchBoundary(pos) {
                if (!Buffer.from(chunck.slice(pos, pos + boundaryBuf.length)).compare(boundaryBuf)) {
                    subInfoArr = [];
                    state = stateSearchFileInfo;
                    return pos + boundaryBuf.length + 2;
                } else {
                    return pos + 1;
                }
            }
            // use chunk, info, data, subInfoArr, state
            function searchFileInfo(pos) {
                if (chunck[pos] == 13 && chunck[pos + 1] == 10 &&
                    chunck[pos + 2] == 13 && chunck[pos + 3] == 10) {
                    // find \r\n\r\n
                    info.push(Buffer.from(subInfoArr).toString());
                    data.length = 0;
                    var tmpFileName = new Date().getTime();
                    tmpFiles.push(tmpFileName);
                    writeStream = fs.createWriteStream(uploadDir + "/" + tmpFileName);
                    state = stateSearchFileEnd;
                    return pos + 4;
                } else {
                    subInfoArr.push(chunck[pos]);
                    return pos + 1;
                }
            }
            // use chunk, writeStream, boundaryBuf, data, subInfoArr, state
            function searchFileEnd(pos) {
                if (chunck[pos] == 13 && chunck[pos + 1] == 10 && chunck[pos + 2] == 45 &&
                    !Buffer.from(chunck.slice(pos + 2, pos + boundaryBuf.length + 2)).compare(boundaryBuf)) {
                    // find boundary, use and to escape unnecessary of produce!!!!!!!!!!!
                    pos += boundaryBuf.length + 2; // \r\nboundary
                    writeStream.write(Buffer.from(data));
                    writeStream.end();
 
                    if (chunck[pos] == 45 && chunck[pos + 1] == 45) { // find "--", end "--\r\n"
                        state = stateSearchBoundary;
                        return pos + 4;
                    } else {  // not end, "\r\n"
                        subInfoArr = [];
                        state = stateSearchFileInfo;
                        return pos + 2;
                    }
                } else {
                    if (data.length > bufSize) {
                        writeStream.write(Buffer.from(data));
                        data.length = 0;
                    }
                    data.push(chunck[pos]);
                    return pos + 1;
                }
            }
            // use chunk, writeStream, info, data, subInfoArr, state
            function reset() {
                writeStream = null, data = [], info = [], subInfoArr = [];
                state = stateSearchBoundary;
                return chunck.length;
            }
 
            var pos = 0;
            while (pos < chunck.length) {
                switch (state) {
                    case stateSearchBoundary: pos = searchBoundary(pos); break;
                    case stateSearchFileInfo: pos = searchFileInfo(pos); break;
                    case stateSearchFileEnd: pos = searchFileEnd(pos); break;
                    default: pos = reset(); callback("Error: parse error", []);
                }
            }
 
        });
        request.on('end', () => {
            var files = [];
            if (info.length != tmpFiles.length) {
                callback("Error: data broken.", []);
                return;
            } else for (var i in tmpFiles) {
                var namePos = info[i].indexOf("name=");
                var filenamePos = info[i].indexOf("filename=");
                var contentTypePos = info[i].indexOf("Content-Type:");
 
                var tmpFileName = tmpFiles[i];
                var name = (namePos != -1) ? info[i].substring(namePos + 6, filenamePos - 3) : "";
                var contentType = (contentTypePos != -1) ? info[i].substring(contentTypePos + 14) : "";
                var filename = (filenamePos != -1) ?
                    info[i].substring(filenamePos + 10, info[i].lastIndexOf("\r") - 1) : "";
 
                files.push({
                    "Content-Type": contentType,
                    "name": name, "filename": filename, "tmpFileName": tmpFileName
                });
            }
            callback("", files);
        });
        request.on('error', (err) => {
            callback("Error: request error," + err.message, []);
        });
    }
}


这段代码的效果好得惊人，甚至可以一次穿好几个linux发行版的安装镜像，也就是说好几个2GB左右大小的ISO通过input multiple写在一个formData里上传，没有一个出错！
