10 文件管理

之前就说过，我们需要的文件管理功能有：文件上传、文件重命名、新建文件夹、身份验证。其实说白了，就一个功能：文件上传。你想，要上传一群文件到服务器，你肯定需要新建一个文件夹再传，如果与服务器上的文件重名，肯定需要对服务器上的文件重命名 或者直接覆盖。为硬盘上日剧韩剧的安全性考虑，我不支持对硬盘上文件内容的改动，包括对已有文件的删除，因此没有写文件删除功能（尽管fs.unlink就可以解决）。

另外，要进行这些操作，你肯定需要有认证权限，为了表面的安全性。对于权限认证，我的思路是这样的：用户输入Key,在服务端编码（例如hex_md5）然后送给服务端，服务端验证正确则发送特定字符表明认证成功。但用户每次操作都要认证则显得有点麻烦，因此必须想个办法，能让认证状态持续一段时间。

为实现这个目的，一开始我使用一个变量存储认证状态，如果为true则是认证了。后来觉得这个想法很不妥，因为客户端的代码，总是有办法改动的，把false改成true就不需要认证了，这不是扯吗？

思考了许久，我决定这样：将加密后的密码保存一段时间，若保存密码的地方为空则重新弹出窗口验证，否则直接将保存起来的密码发送给服务端验证。这样，从用户的角度，便避免了重复验证。

这里主要说服务端的原理。文件上传只需要利用之前写好的表单文件处理函数即可。文件重命名可以使用fs.rename, 新建文件夹使用fs.mkdir,因此代码很简单：

/*
 * use fs, formFileParser(written by Mzero for MikoSite, MIT Licence), http(Request, Response)
 * written by Mzero for MikoSite, MIT License
 * recv Action(auth, mkdir, mv), recv File  
 * if no md5 module, it will use it's own simple encode func
 */
var fs = require('fs');
var formFileParser = require('./formFileParser');
 
module.exports = function recvData(opts) {
    // parse Parameter ============================================//
    if (!opts) return null;
    if (!("key" in opts)) opts.key = "MikoSite";
    if (!("encode" in opts)) opts.encode = easyEncode; // localEncode function
 
    var authKey = opts.encode(opts.key);
 
    // Independent =======================================//
    function easyEncode(code) {
        var c = String.fromCharCode(code.charCodeAt(0) + code.length);
        for (var i = 1; i < code.length; i++)
            c += String.fromCharCode(code.charCodeAt(i) + code.charCodeAt(i - 1));
        return escape(c);
    }
 
    function getRealPath(realDataPath, subDataPath, urlPath) {
        var subPath = "";
        var pathList = urlPath.split('/').filter(function (s) { return s && s.trim(); });
        if (!pathList.length) {
            subPath = "/";
        } else for (var i = 0; i < pathList.length; i++) {
            subPath += "/" + decodeURIComponent(pathList[i].toString());
        }
        return realDataPath + subPath.substring(subDataPath.length);
    }
 
    function auth(authKey, content, Response) {
        if (content.key == authKey)
            Response.end("auth pass" + content.key);
        else
            Response.end("auth fail" + content.key);
        // console.log(content, content.key, authKey);
    }
 
    function mkdir(realDataPath, subDataPath, content, Response) {
        var pathname = getRealPath(realDataPath, subDataPath, content.path)
        var name = pathname.substring(pathname.lastIndexOf("/") + 1);
        fs.exists(pathname, function (err) {
            if (err) {
                Response.end("dir: " + name + " exist.");
                return;
            }
            fs.mkdir(pathname, { recursive: true }, function (err) {
                if (err) {
                    Response.end("mkdir : " + name + " fail");
                    return;
                }
                Response.end("mkdir : " + name + " success");
                console.log("mkdir : ", pathname);
            });
        });
    }
 
    function rename(realDataPath, subDataPath, content, Response) {
        var oldPathname = getRealPath(realDataPath, subDataPath, content.oldPath);
        var newPathname = getRealPath(realDataPath, subDataPath, content.newPath);
        var oldName = oldPathname.substring(oldPathname.lastIndexOf("/") + 1);
        var newName = newPathname.substring(newPathname.lastIndexOf("/") + 1);
        fs.exists(newPathname, function (err) {
            if (err) {
                Response.end("file: " + newName + " already exist.");
                return;
            }
            fs.rename(oldPathname, newPathname, function (err) {
                if (err) {
                    Response.end("rename: " + oldName + " > " + newName + " fail");
                    return;
                }
                Response.end("rename: " + oldName + " > " + newName + " success");
                console.log("rename: " + oldPathname + " > " + newPathname);
            });
        });
 
    }
 
    function mvUploadFile(path, oriName, newName, Response) {
        fs.stat(path + "/" + newName, function (err, stats) {
            if (err) {
                fs.rename(path + "/" + oriName, path + "/" + newName, function (err) {
                    if (err) {
                        Response.end("upload: fail, can not rename");
                        console.log("upload:", path + "/" + oriName, "with original fileName:", newName);
                        // try { fs.unlink(path + "/" + oriName, (err) => { }); } catch (err) { }
                        return;
                    }
                    Response.end("upload: success")
                    console.log("upload:", path + "/" + newName);
                });
            } else {
                Response.end("upload: fail, file exists already.");
                try { fs.unlink(path + "/" + oriName, (err) => { }); } catch (err) { }
            }
        });
    }
 
    // Methods============================================//
    // using authKey
    this.parseAction = function (realDataPath, subDataPath, Request, Response) {
        var postStr = "";
        Request.on("data", function (chunk) { postStr += chunk; });
        Request.on("end", function () {
            postStr = JSON.parse(postStr);
            // console.log(postStr);
            switch (postStr.action) {
                case "auth":
                    auth(authKey, JSON.parse(postStr.content), Response);
                    break;
                case "mkdir":
                    mkdir(realDataPath, subDataPath, JSON.parse(postStr.content), Response);
                    break;
                case "rename":
                    rename(realDataPath, subDataPath, JSON.parse(postStr.content), Response);
                    break;
            }
        });
    }
    // using formFileParser
    this.parseUpload = function (realPath, Request, Response) {
        var formfile = new formFileParser({ uploadDir: realPath });
        if (!formfile) { Response.end("upload: fail"); return; }
        formfile.parse(Request, function (err, files) {
            try {
                for (var i in files) {
                    mvUploadFile(realPath, files[i].tmpFileName, files[i].filename, Response);
                }
            } catch (err) {
                // console.log("no file");
            }
        });
    }
}


11 写一个和”app.get(url, callback);” 类似的东西

功能多了之后，不同的功能不同标记，管理起来就很麻烦。我之前用switch case分别判断每种pattern, 符合哪种，代入那种，多了之后代码就很不美观。express常用app.get()/app.post() 来在url这里描述一种模式，比如：

app.get(“/mShare/*”,(req,res,next)=>{...});

就是表示请求/mShare/.... 都能被这条匹配上，前提是它之前没有规定相同的路径。基于这条特性，我也想写个模仿一下，至少比switch case优雅。

由于我没有什么复杂的/mShare/*/a/*/bn/*/upload这种路径要匹配，因此只需要关注开头匹不匹配就行（即/mShare/*类型）。因此有上述代码。那么来了请求该怎么办呢？

    var specList = [];
    function parseSpec(Request, Response) {
        // subPath: make sure to get "/, /mShare ...", with all URI decoded
        // urlPath: make sure to get "/, /ab%20cd ...", with all URI encoded(original)
        var subPath = "", urlPath = "";
        var tmp = url.parse(Request.url).pathname;
        var oriPathList = tmp.split(/\//).filter(function (s) { return s && s.trim(); });
        var pathList = []; // get clean pathList without ..
        for (var i = 0; i < oriPathList.length; i++) {
            if (oriPathList[i].trim() == "..") pathList.pop();
            else pathList.push(oriPathList[i]);
        }
        if (!pathList.length) {
            subPath = "/";
            urlPath = "/";
        } else for (var i = 0; i < pathList.length; i++) {
            subPath += "/" + decodeURIComponent(pathList[i].toString());
            urlPath += "/" + pathList[i].toString();
        }
        // match Spec requirements 
        for (var i in specList) {
            var headPath = specList[i].headPath;
            if (headPath == "/") headPath = "";
            var urlHeadPath = specList[i].urlHeadPath;
            var callback = specList[i].callback;
            if (subPath.indexOf(headPath) == 0) {
                // get realPath and urlPath without the headPath
                var restPath = "", urlRestPath = "";
                restPath = subPath.substring(headPath.length);
                urlRestPath = urlPath.substring(urlHeadPath.length);
                // console.log(subPath, urlPath, callback.name);
                // console.log("restPath:",restPath, "urlRestPath:",urlRestPath);
                callback(restPath, urlRestPath, Request, Response);
                return;
            }
        }
        // default
        Response.statusCode = 200;

        Response.write('top:50%;left:50%;transform:translate(-50%,-50%);position:fixed;">');
        Response.end('Service Error</div>');
    }
