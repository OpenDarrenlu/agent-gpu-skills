/**
 * Note: The returned array must be malloced, assume caller calls free().
 */
#define MAX_LEN 100000

struct hashStr {
    char key[11];
    int num;
    UT_hash_handle hh;
};

struct hashStr *g_user = NULL;

char ** findRepeatedDnaSequences(char * s, int* returnSize){
    g_user = NULL;
    int n = strlen(s);
    char ** res = (char **)malloc(sizeof(char*) * MAX_LEN);
    *returnSize = 0;
    char tempStr[11];
    tempStr[10] = '\0';
    for (int i = 0; i < n - 9; i++) {
        strncpy(tempStr, &s[i], 10);
        struct hashStr *s;
        HASH_FIND_STR(g_user, &tempStr, s);
        if (s == NULL) {
            s = (struct hashStr*)malloc(sizeof(struct hashStr));
            strcpy(s->key, tempStr);
            s->num = 1;
            HASH_ADD_STR(g_user, key, s);
        } else {
            s->num++;
            if (s->num == 2) {
                res[*returnSize] = (char*)malloc(sizeof(char) * 11);
                strcpy(res[*returnSize], tempStr);
                (*returnSize)++;
            }
        }
    }
    return res;
}
