5 读取目录

想要发送目录结构，首先就应该会读取目录信息。读取目录下的文件由fs.readdir 完成，读取目录下的文件的信息由对每个文件进行fs.stat完成。根据这个思路，我们可以快速写出代码：

function readIndex(realPath, isFileSign, IconSet, urlPath, Index, Response) {
        try {
            if (!fs.statSync(realPath).isDirectory()) {
                Response.end(isFileSign);
                return;
            }
        } catch (err) {
            send404(Response);
            return;
        }
        fs.readdir(realPath, function (err, files) {
            // add item in fileList
            var fileList = [];
            for (var i in files) {
                try {
                    var filestats = fs.statSync(realPath + '/' + files[i]);
                } catch (err) {
                    continue;
                }
                var icon = filestats.isDirectory() ?
                    IconSet.Folder : IconChoose(realPath + '/' + files[i], IconSet);
                var time = filestats.mtime.toJSON().split('T')[0];
                time = time + ' ' + filestats.mtime.toString().split(' ')[4];
                fileList.push({ "icon": icon, "name": files[i], "time": time, "size": filestats.size });
            }
            // check if have webPage 
            var webPage = "";
            for (var i in fileList) {
                for (var j in Index) {
                    if (fileList[i].name == IndexList[j]) {
                        webPage = urlPath + "/" + IndexList[j];
                        break;
                    }
                }
            }
            // send info 
            var info = {
                "webPage": webPage,
                "IconBlank": IconSet.Blank,
                "IconHome": IconSet.Home,
                "fileList": fileList
            };
            Response.write(JSON.stringify(info));
            Response.end();
        });
}


这样，我们实现了读取目录文件 文件名、修改日期、大小、适用的图片（代码与TypeChoose类似）记录到info.fileList、记录目录中是否有符合诸如index.html index.htm (在Index中定义)的文件记录到info对象中，并发送info对象。

从逻辑上讲，代码并没有问题，但从效率上讲，涉及磁盘的代码一般等待时间较长，在for循环内不断执行fs.stat效率很低，因此加载时间会很长。因此，我们改进为异步的方法，每次调用异步fs.stat函数，回调函数中将信息加入fileList数组。

可这样又带来问题，我怎么知道所有文件的信息都已经读取完成了呢？ 一个简单的思路就是比较fs.readdir返回数组的长度和当前fileList的长度，如果一样长，那么便读取完成。

可是fs.stat也会有出错的时候，比如当读取的文件是软链接但软链接失效的时候。那，这也简单，记录错误的次数，比较错误次数与fileList的和与fs.readdir返回数组的长度是否一致就好了。

假设有2400个文件，每次都要做加法然后比较，速度较慢。我们可以给错误次数分类。如果其为0（这是大多数情况），只需比较fileList 和 fs.readdir返回数组的长度是否一致即可。

那么这个比较过程该放在哪呢？

一开始，由于写法习惯，我写成了这样：

var files = fs.readdirSync(realPath);
if (!files.length) {
    sendInfoCallBack(info); return;
}
var errTimes = 0; // if err, it show be taken into account to indicate the end
files.forEach((name) => {
    fs.stat(realPath + '/' + name, (err, stats) => {
        if (err) {
            errTimes++;
            return;
        }
        ...
        var stopSign = (!errTimes && info.fileList.length == files.length);
        stopSign = stopSign || (errTimes && errTimes + info.fileList.length == files.length);
        if (stopSign) sendInfoCallBack(info);

看似并没有什么问题，其实暗藏危机：如果恰好读到错误得文件在末尾，那么stopSign永远不会为1——因为在最后一个fs.stat执行完时，由于发生错误，回调函数退出，stopSign没有变成1，sendInfoCallBack(info)永远没被执行，进而导致客户端没有收到response而阻塞。

改进的方法很简单，只要让stopSign的判断在发生错误时也执行即可。最终，代码如下所示：

function readIndex(realPath, urlPath, exploreMode, sendInfoCallBack) {
    var info = {
        exploreMode: exploreMode, IconHome: IconHome, currentUrlPath: urlPath,
        webPage: { rank: -1, path: "" }, fileList: []
    };
    try {
        var files = fs.readdirSync(realPath);
        if (!files.length) {
            sendInfoCallBack(info); return;
        }
        var errTimes = 0; // if err, it show be taken into account to indicate the end
        files.forEach((name) => {
            fs.stat(realPath + '/' + name, (err, stats) => {
                if (err) {
                    errTimes++;
                } else {
                    var icon = stats.isDirectory() ? IconFolder : IconChoose(realPath + '/' + name);
                    var time = stats.mtime.toJSON().split('T')[0];
                    time = time + ' ' + stats.mtime.toString().split(' ')[4];
                    info.fileList.push({ icon: icon, name: name, time: time, size: stats.size });
                    // check if have webPage, make sure considering index.html first;
                    if (!exploreMode) {
                        var rank = IndexList.indexOf(name);
                        if (rank != -1 && (info.webPage.rank == -1 || rank < info.webPage.rank)) {
                            info.webPage.rank = rank;
                            if (urlPath.substring(urlPath.length - 1) == "/") {
                                urlPath = urlPath.substring(0, urlPath.length - 1);
                            }
                            info.webPage.path = urlPath + "/" + encodeURIComponent(name);
                        }
                    }
                }
                var stopSign = (!errTimes && info.fileList.length == files.length);
                stopSign = stopSign || (errTimes && errTimes + info.fileList.length == files.length);
                if (stopSign) sendInfoCallBack(info);
            });
        });
    } catch (err) {
        // console.log(err);
        info.webPage.rank = -2;
        sendInfoCallBack(info);
    }
}


其中，exploreMode表示是否重定向到目录下的index.html，若为true则否。WebPage.rank 用于判断是否为最优的结果。-1为初值，-2表示出错，0表示匹配到IndexList的第0项。例如，IndexList为[“index.html”, “index.htm”],若目录下同时出现两个文件，我们希望重定向到index.html而不是index.htm,此时rank就起作用了。

6 发送目录

在5中，我们已经可以读取一个目录的信息并将其存在一个对象里等待sendInfoCallBack函数的调用了。那么现在问题来了：我们怎么把这个对象变成我们可见的网页呢？

模板，模板可以做到这个事情。但我们的目标是原生，因此尽量不使用外部模块。目前，我想到的方法是将数据写成<script>var __INFO = info;</script>的形式，先发送出去，然后读取html文件发送，在html文件中，我们便可以使用info了。

代码如下：

function sendIndex(info, Response) {
        Response.statusCode = 200;
        Response.setHeader('Content-Type', 'text/html; charset=utf-8');

7 整合 发送文件和目录

为最终实现Apache服务器那样的效果，我们需要将文件读取和目录读取整合到一起，从而实现点击文件则读文件，点击目录则读目录的效果。

整合的代码如下所示：

this.sendData = (realPath, urlPath, exploreMode, insideMode, Request, Response) => {
        sendFile(realPath, Request, Response, (err) => {
            if (!err) return;
            // console.time("DirStat");
            readIndex(realPath, urlPath, exploreMode, (info) => {
                // console.timeEnd("DirStat");
                try {
                    if (info.webPage.rank == -2) {
                        send404(Response);
                    } else if (info.webPage.rank != -1) {
                        reDirect(301, info.webPage.path, Request, Response);
                    } else if (insideMode) {
                        Response.statusCode = 200;
                        Response.setHeader('Content-Type', 'text/html; charset=utf-8');
                        Response.end(JSON.stringify(info));
                    } else {
                        sendIndex(info, Response);
                    }
                } catch (err) {
                    // console.log(err);
                }
            });
        });
    }


其处理的内容其实在写读取目录这一小节的时候已经提及，其中insideMode只发送数据本身，不发送html文件。其核心便是sendFile 和 readIndex这两个函数。处理逻辑为：如果读取文件失败，表明其可能为目录，交给readIndex处理，readIndex将处理结果交给回调函数判断是否为错误（发送404），是否需要重定向，是否需要直接发送数据，否则默认使用sendIndex发送数据和对应的html文档。
